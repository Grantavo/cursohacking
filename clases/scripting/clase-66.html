  local results

  # Convertir JSON a lГғВӯneas delimitadas
  results=$(echo "$json" | jq -c '.results[]' 2>/dev/null)

  if [[ -n "$NAME" ]]; then
    results=$(echo "$results" | while read -r m; do
      local name; name=$(echo "$m" | jq -r '.name')
      if echo "$name" | grep -qi "$NAME"; then
        echo "$m"
      fi
    done)
  fi

  if [[ -n "$OS" ]]; then
    results=$(echo "$results" | while read -r m; do
      local os; os=$(echo "$m" | jq -r '.os')
      if echo "$os" | grep -qi "$OS"; then
        echo "$m"
      fi
    done)
  fi

  if [[ -n "$DIFFICULTY" ]]; then
    results=$(echo "$results" | while read -r m; do
      local diff; diff=$(echo "$m" | jq -r '.difficulty_text')
      if echo "$diff" | grep -qi "$DIFFICULTY"; then
        echo "$m"
      fi
    done)
  fi

  echo "$results"
}
