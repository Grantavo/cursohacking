Scripting en Bash [4-6]

Duración: 20:05 · Proyecto 1: Buscador HTB

Mejoramos la salida del script: tablas con columnas perfectamente alineadas, colores asignados por dificultad y paginación de resultados para no saturar la terminal.

Colores por dificultad

Cada nivel de dificultad tiene un color distintivo:

get_difficulty_color() {
  local diff="$1"
  case "${diff,,}" in
    easy)   echo "$GREEN" ;;
    medium) echo "$YELLOW" ;;
    hard)   echo "$RED" ;;
    insane) echo "$MAGENTA" ;;
    *)      echo "$RESET" ;;
  esac
}

format_difficulty() {
  local diff="$1"
  local color; color=$(get_difficulty_color "$diff")
  local icon
  case "${diff,,}" in
    easy)   icon="●" ;;
    medium) icon="●" ;;
    hard)   icon="●" ;;
    insane) icon="●" ;;
    *)      icon="○" ;;
  esac
  echo "${color}${icon} ${diff^}${RESET}"
}

Tabla con columnas alineadas

Usamos printf con ancho fijo para alinear perfectamente:

print_table_header() {
  local cols=("Nombre" "IP" "SO" "Dificultad" "Puntos")
  local widths=(24 16 10 14 8)

  # Línea superior
  echo -n "┌"
  for w in "${widths[@]}"; do echo -n "$(printf '─%.0s' $(seq 1 $((w+2))))"; [[ $w != "${widths[-1]}" ]] && echo -n "┬"; done
  echo "┐"

  # Cabeceras
  echo -n "│"
  for i in "${!cols[@]}"; do
    printf " %-${widths[$i]}s │" "${cols[$i]}"
  done
  echo

  # Separador
  echo -n "├"
  for w in "${widths[@]}"; do echo -n "$(printf '─%.0s' $(seq 1 $((w+2))))"; [[ $w != "${widths[-1]}" ]] && echo -n "┼"; done
  echo "┤"
}

print_table_footer() {
  local widths=(24 16 10 14 8)
  echo -n "└"
  for w in "${widths[@]}"; do echo -n "$(printf '─%.0s' $(seq 1 $((w+2))))"; [[ $w != "${widths[-1]}" ]] && echo -n "┴"; done
  echo "┘"
}

Paginación de resultados

Cuando hay muchos resultados, mostramos de 10 en 10:

paginate_results() {
  local results=("$@")
  local page=0
  local page_size=10
  local total=${#results[@]}
  local pages=$(( (total + page_size - 1) / page_size ))

  while true; do
    clear
    print_banner
    print_table_header

    local start=$((page * page_size))
    local end=$((start + page_size))
    [[ $end -gt $total ]] && end=$total

    for ((i = start; i < end; i++)); do
      IFS='|' read -r name ip os diff points <<< "${results[$i]}"
      local color; color=$(get_difficulty_color "$diff")
      printf "│ %-24s │ %-16s │ %-10s │ ${color}%-14s${RESET} │ %-8s │\n" "$name" "$ip" "$os" "${diff^}" "$points"
    done

    print_table_footer
    echo -e "\n${BLUE}Página $((page+1))/$pages${RESET} | Mostrando $((end-start)) de $total resultados"

    if [[ $pages -gt 1 ]]; then
      echo -e "\n${YELLOW}[N] Siguiente  [P] Anterior  [Q] Salir${RESET}"
      read -r -n 1 key
      case "$key" in
        n|N) [[ $page -lt $((pages-1)) ]] && ((page++)) ;;
        p|P) [[ $page -gt 0 ]] && ((page--)) ;;
        q|Q) break ;;
      esac
    else
      break
    fi
  done
}
💡 Paginación interactiva: read -n 1 captura una sola tecla sin necesidad de pulsar Enter. Es la forma estándar de hacer menús interactivos en Bash.

Actualización de render_results

Integramos paginación en el flujo principal:

render_results() {
  local results="$1"
  local lines=()

  while read -r machine; do
    [[ -z "$machine" ]] && continue
    local name ip os diff points
    name=$(echo "$machine" | jq -r '.name')
    ip=$(echo "$machine" | jq -r '.ip // "N/A"')
    os=$(echo "$machine" | jq -r '.os')
    diff=$(echo "$machine" | jq -r '.difficulty_text // "unknown"')
    points=$(echo "$machine" | jq -r '.points // 0')
    lines+=("${name}|${ip}|${os}|${diff}|${points}")
  done <<< "$results"

  if [[ ${#lines[@]} -eq 0 ]]; then
    print_error "No se encontraron resultados"
    return
  fi

  paginate_results "${lines[@]}"
}
⚠️ Compatibilidad: Los caracteres de línea de caja (┌┬┐├┼┤└┴┘) requieren UTF-8. Asegúrate de que tu terminal lo soporte con echo $LANG.
