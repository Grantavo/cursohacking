Scripting en Bash [5-6]

Duración: 22:12 · Proyecto 1: Buscador HTB

Mejoramos el script con sistema de caché para evitar llamadas repetidas a la API, medición de tiempo de respuesta, manejo robusto de errores de red, integración con VPN y un modo interactivo con read y select.

Sistema de caché

Guardamos las respuestas de la API en archivos temporales con TTL configurable:

get_cache_path() {
  mkdir -p "$CACHE_DIR"
  echo "${CACHE_DIR}/machines.json"
}

is_cache_valid() {
  local cache_file; cache_file=$(get_cache_path)
  [[ -f "$cache_file" ]] || return 1
  local now ttl_age
  now=$(date +%s)
  ttl_age=$((now - CACHE_TTL))
  [[ $(stat -c %Y "$cache_file") -gt $ttl_age ]]
}

fetch_with_cache() {
  local cache_file; cache_file=$(get_cache_path)

  if is_cache_valid; then
    print_info "Usando caché (TTL: ${CACHE_TTL}s)"
    cat "$cache_file"
    return 0
  fi

  print_info "Consultando API..."
  local json
  json=$(fetch_machines) || return 1
  echo "$json" > "$cache_file"
  print_success "Caché actualizada"
  echo "$json"
}

Tiempo de respuesta

Medimos cuánto tarda la petición con time interno:

fetch_with_timing() {
  local start end duration
  start=$(date +%s%N)

  local json
  json=$(fetch_with_cache) || return 1

  end=$(date +%s%N)
  duration=$(( (end - start) / 1000000 ))

  if [[ $duration -gt 1000 ]]; then
    print_info "Tiempo de respuesta: ${YELLOW}${duration}ms${RESET}"
  else
    print_info "Tiempo de respuesta: ${GREEN}${duration}ms${RESET}"
  fi

  echo "$json"
}

Manejo de errores de red

Detectamos y reportamos errores de conexión específicos:

check_connectivity() {
  local test_url="https://www.hackthebox.com"
  print_info "Verificando conexión a ${test_url}..."

  if ! curl -s --connect-timeout 5 "$test_url" > /dev/null 2>&1; then
    print_error "Sin conexión a internet o HTB está caído"
    echo "  Posibles causas:"
    echo "  • No tienes conexión a internet"
    echo "  • HTB está temporalmente caído"
    echo "  • Tu VPN está interfiriendo"
    return 1
  fi

  print_success "Conexión OK"
  return 0
}

Conexión VPN

Detectamos si estamos conectados a la VPN de HTB:

check_vpn() {
  local vpn_ip
  vpn_ip=$(ip a show tun0 2>/dev/null | grep -oP 'inet \K[\d.]+')

  if [[ -n "$vpn_ip" ]]; then
    print_success "VPN conectada: ${vpn_ip}"
    return 0
  else
    print_info "VPN no detectada (interfaz tun0)"
    echo "  Algunas máquinas solo son accesibles vía VPN."
    echo "  Conéctate con: sudo openvpn ~/Downloads/your_file.ovpn"
    return 1
  fi
}

Modo interactivo

Usamos select para crear un menú de opciones:

interactive_mode() {
  print_banner
  echo "${BOLD}Modo interactivo${RESET}"
  echo

  # Verificar conexión primero
  check_connectivity || return 1

  # Verificar VPN
  check_vpn

  while true; do
    echo -e "\n${BOLD}¿Qué deseas hacer?${RESET}"
    select opt in "Buscar por nombre" "Filtrar por SO" "Filtrar por dificultad" "Ver todas" "Salir"; do
      case $opt in
        "Buscar por nombre")
          read -r -p "Nombre de la máquina: " NAME
          do_search
          break
          ;;
        "Filtrar por SO")
          select OS in "Linux" "Windows" "FreeBSD"; do
            [[ -n "$OS" ]] && break
          done
          do_search
          break
          ;;
        "Filtrar por dificultad")
          select DIFFICULTY in "easy" "medium" "hard" "insane"; do
            [[ -n "$DIFFICULTY" ]] && break
          done
          do_search
          break
          ;;
        "Ver todas")
          do_search
          break
          ;;
        "Salir")
          print_info "¡Hasta luego!"
          exit 0
          ;;
        *) echo "Opción inválida" ;;
      esac
    done
  done
}
💡 read -r vs select: read captura texto libre; select presenta un menú numerado y captura la selección automáticamente. Ambos son esenciales para scripts interactivos.
