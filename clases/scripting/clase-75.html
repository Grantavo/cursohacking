Scripting en Bash [5-15]

Duración: 31:24 · Proyecto 2: Ruleta de casino

Implementamos el historial de jugadas usando arrays, estadísticas como números más frecuentes y racha de colores, y mostramos toda esta información al jugador.

Arrays para guardar resultados

Cada jugada se guarda como un string delimitado en un array:

declare -a HISTORIAL=()
TOTAL_JUGADAS=0
TOTAL_GANADAS=0
TOTAL_PERDIDAS=0
MAX_SALDO=$SALDO

registrar_jugada() {
  local numero=$1
  local color=$2
  local paridad=$3
  local ganancia=$4
  local apuesta=$5

  local resultado="perdida"
  [[ "$ganancia" -gt 0 ]] && resultado="ganada"

  HISTORIAL+=("${numero}:${color}:${paridad}:${apuesta}:${ganancia}:${resultado}")

  (( TOTAL_JUGADAS++ ))
  if [[ "$ganancia" -gt 0 ]]; then
    (( TOTAL_GANADAS++ ))
  else
    (( TOTAL_PERDIDAS++ ))
  fi

  [[ "$SALDO" -gt "$MAX_SALDO" ]] && MAX_SALDO=$SALDO
}

Estadísticas: números más frecuentes

Usamos un array asociativo para contar frecuencias:

numeros_mas_frecuentes() {
  declare -A frecuencias

  for jugada in "${HISTORIAL[@]}"; do
    local num
    num=$(echo "$jugada" | cut -d: -f1)
    (( frecuencias[$num]++ ))
  done

  # Ordenar por frecuencia (usando sort externo)
  echo "Números más frecuentes:"
  for num in "${!frecuencias[@]}"; do
    echo "$num:${frecuencias[$num]}"
  done | sort -t: -k2 -rn | head -5 | while IFS=: read -r num freq; do
    printf "  Número %2d: %d veces\n" "$num" "$freq"
  done
}

Estadísticas: racha de colores

racha_actual() {
  local color_objetivo=$1
  local racha=0

  for (( i = ${#HISTORIAL[@]} - 1; i >= 0; i-- )); do
    local color
    color=$(echo "${HISTORIAL[$i]}" | cut -d: -f2)
    if [[ "$color" == "$color_objetivo" ]]; then
      (( racha++ ))
    else
      break
    fi
  done

  echo "$racha"
}

mayor_racha() {
  local color_objetivo=$1
  local max_racha=0
  local racha_actual=0

  for jugada in "${HISTORIAL[@]}"; do
    local color
    color=$(echo "$jugada" | cut -d: -f2)
    if [[ "$color" == "$color_objetivo" ]]; then
      (( racha_actual++ ))
      [[ "$racha_actual" -gt "$max_racha" ]] && max_racha=$racha_actual
    else
      racha_actual=0
    fi
  done

  echo "$max_racha"
}

Mostrar historial

Visualizamos las últimas jugadas en una tabla:

mostrar_historial() {
  echo "${BOLD}Últimas jugadas:${RESET}"
  echo

  local start=0
  local limit=20  # Mostrar últimas 20
  local total=${#HISTORIAL[@]}

  if [[ "$total" -gt "$limit" ]]; then
    start=$(( total - limit ))
  fi

  printf "%-4s %-6s %-8s %-8s %-10s\n" "#" "Núm" "Color" "Apuesta" "Resultado"
  echo "----------------------------------------"

  for (( i = start; i < total; i++ )); do
    IFS=: read -r num color paridad apuesta ganancia resultado <<< "${HISTORIAL[$i]}"

    local color_display
    case "$color" in
      rojo)  color_display="${ROJO}$color${RESET}" ;;
      negro) color_display="${BLANCO}$color${RESET}" ;;
      verde) color_display="${VERDE}$color${RESET}" ;;
    esac

    local result_display
    if [[ "$resultado" == "ganada" ]]; then
      result_display="${VERDE}✓ +\$$ganancia${RESET}"
    else
      result_display="${ROJO}✗ -\$$apuesta${RESET}"
    fi

    printf "%-4d %-6s %b %-8s %b\n" "$((i+1))" "$num" "$color_display" "\$$apuesta" "$result_display"
  done
}

Panel de estadísticas completo

mostrar_estadisticas() {
  clear
  echo "${BOLD}${AZUL}═══════════════════════════════════${RESET}"
  echo "${BOLD}         📊 ESTADÍSTICAS${RESET}"
  echo "${BOLD}${AZUL}═══════════════════════════════════${RESET}"
  echo

  if [[ "$TOTAL_JUGADAS" -eq 0 ]]; then
    echo "Aún no hay jugadas registradas."
    return
  fi

  local winrate
  winrate=$(echo "scale=2; $TOTAL_GANADAS * 100 / $TOTAL_JUGADAS" | bc 2>/dev/null || echo "0")
  local racha_rojo
  racha_rojo=$(racha_actual "rojo")
  local racha_negro
  racha_negro=$(racha_actual "negro")

  echo "Jugadas totales:    $TOTAL_JUGADAS"
  echo "Ganadas:            ${VERDE}$TOTAL_GANADAS${RESET}"
  echo "Perdidas:           ${ROJO}$TOTAL_PERDIDAS${RESET}"
  echo "Winrate:            ${AMARILLO}${winrate}%${RESET}"
  echo "Saldo actual:       ${VERDE}\$${SALDO}${RESET}"
  echo "Saldo máximo:       ${VERDE}\$${MAX_SALDO}${RESET}"
  echo "Racha rojo actual:  ${ROJO}$racha_rojo${RESET}"
  echo "Racha negro actual: ${BLANCO}$racha_negro${RESET}"
  echo
  numeros_mas_frecuentes
  echo
  mostrar_historial

  read -n 1 -s -r -p "Presiona cualquier tecla para continuar..."
}
💡 bc para decimales: Bash solo soporta aritmética entera. Para el winrate (porcentaje con decimales) usamos bc, una calculadora de precisión arbitraria.
