Scripting en Bash [6-15]

Duración: 15:24 · Proyecto 2: Ruleta de casino

Añadimos persistencia al juego: guardar la partida en un archivo, cargarla al iniciar, usar formato JSON con jq y mantener el saldo y las estadísticas entre sesiones.

Guardar partida

Exportamos el estado del juego a un archivo JSON:

GUARDAR_DIR="${HOME}/.roulette"
GUARDAR_FILE="${GUARDAR_DIR}/save.json"

guardar_partida() {
  mkdir -p "$GUARDAR_DIR"

  # Construir JSON manualmente (o con jq si está disponible)
  local json="{"

  json+="\"nombre\": \"${NOMBRE}\","
  json+="\"saldo\": ${SALDO},"
  json+="\"max_saldo\": ${MAX_SALDO},"
  json+="\"total_jugadas\": ${TOTAL_JUGADAS},"
  json+="\"total_ganadas\": ${TOTAL_GANADAS},"
  json+="\"total_perdidas\": ${TOTAL_PERDIDAS},"
  json+="\"historial\": ["

  local first=true
  for jugada in "${HISTORIAL[@]}"; do
    IFS=: read -r num color paridad apuesta ganancia resultado <<< "$jugada"
    [[ "$first" == false ]] && json+=","
    first=false
    json+="{"
    json+="\"numero\": ${num},"
    json+="\"color\": \"${color}\","
    json+="\"paridad\": \"${paridad}\","
    json+="\"apuesta\": ${apuesta},"
    json+="\"ganancia\": ${ganancia},"
    json+="\"resultado\": \"${resultado}\""
    json+="}"
  done

  json+="]}"

  echo "$json" > "$GUARDAR_FILE"
  print_success "Partida guardada en ${GUARDAR_FILE}"
}

Cargar partida guardada

Usamos jq para parsear el JSON si está disponible:

cargar_partida() {
  if [[ ! -f "$GUARDAR_FILE" ]]; then
    print_info "No hay partida guardada"
    return 1
  fi

  print_info "Cargando partida guardada..."

  if command -v jq &> /dev/null; then
    NOMBRE=$(jq -r '.nombre' "$GUARDAR_FILE")
    SALDO=$(jq -r '.saldo' "$GUARDAR_FILE")
    MAX_SALDO=$(jq -r '.max_saldo' "$GUARDAR_FILE")
    TOTAL_JUGADAS=$(jq -r '.total_jugadas' "$GUARDAR_FILE")
    TOTAL_GANADAS=$(jq -r '.total_ganadas' "$GUARDAR_FILE")
    TOTAL_PERDIDAS=$(jq -r '.total_perdidas' "$GUARDAR_FILE")

    HISTORIAL=()
    while read -r jugada; do
      local num color paridad apuesta ganancia resultado
      num=$(echo "$jugada" | jq -r '.numero')
      color=$(echo "$jugada" | jq -r '.color')
      paridad=$(echo "$jugada" | jq -r '.paridad')
      apuesta=$(echo "$jugada" | jq -r '.apuesta')
      ganancia=$(echo "$jugada" | jq -r '.ganancia')
      resultado=$(echo "$jugada" | jq -r '.resultado')
      HISTORIAL+=("${num}:${color}:${paridad}:${apuesta}:${ganancia}:${resultado}")
    done <<< "$(jq -c '.historial[]' "$GUARDAR_FILE")"
  else
    # Fallback: leer línea por línea (formato simple)
    cargar_partida_simple
  fi

  print_success "Partida cargada. Bienvenido de nuevo, ${NOMBRE}!"
}

Formato JSON con jq

Si jq está instalado, también podemos usarlo para construir el JSON de forma más limpia:

guardar_partida_jq() {
  mkdir -p "$GUARDAR_DIR"

  # Construir array de historial
  local historial_json="[]"
  for jugada in "${HISTORIAL[@]}"; do
    IFS=: read -r num color paridad apuesta ganancia resultado <<< "$jugada"
    local item
    item=$(jq -n \
      --argjson n "$num" \
      --arg c "$color" \
      --arg p "$paridad" \
      --argjson a "$apuesta" \
      --argjson g "$ganancia" \
      --arg r "$resultado" \
      '{numero: $n, color: $c, paridad: $p, apuesta: $a, ganancia: $g, resultado: $r}')
    historial_json=$(echo "$historial_json" | jq --argjson item "$item" '. + [$item]')
  done

  jq -n \
    --arg nombre "$NOMBRE" \
    --argjson saldo "$SALDO" \
    --argjson max_saldo "$MAX_SALDO" \
    --argjson total "$TOTAL_JUGADAS" \
    --argjson ganadas "$TOTAL_GANADAS" \
    --argjson perdidas "$TOTAL_PERDIDAS" \
    --argjson historial "$historial_json" \
    '{nombre: $nombre, saldo: $saldo, max_saldo: $max_saldo,
      total_jugadas: $total, total_ganadas: $ganadas,
      total_perdidas: $perdidas, historial: $historial}' > "$GUARDAR_FILE"

  print_success "Partida guardada con formato JSON"
}
💡 jq -n para construir JSON desde cero: La opción -n (null input) permite crear objetos JSON sin necesidad de pipear datos de entrada. Combinado con --arg y --argjson es muy potente.

Persistencia de saldo y estadísticas

Integramos la carga al inicio del juego:

iniciar_juego() {
  # Preguntar si cargar partida
  if [[ -f "$GUARDAR_FILE" ]]; then
    read -r -p "¿Cargar partida guardada? (s/N): " respuesta
    if [[ "${respuesta,,}" == "s" ]]; then
      cargar_partida && return 0
    fi
  fi

  # Si no se carga, empezar de nuevo
  bienvenida
  read -r -p "Introduce tu nombre: " NOMBRE
  NOMBRE="${NOMBRE:-Jugador}"
  read -r -p "Saldo inicial [1000]: " saldo_in
  SALDO="${saldo_in:-1000}"
  MAX_SALDO=$SALDO
  TOTAL_JUGADAS=0
  TOTAL_GANADAS=0
  TOTAL_PERDIDAS=0
  HISTORIAL=()
}
⚠️ jq como dependencia: El script funciona sin jq usando fallbacks. Pero para una experiencia completa, recomendamos instalarlo: sudo apt install jq.
