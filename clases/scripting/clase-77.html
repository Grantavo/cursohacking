Scripting en Bash [7-15]

Duración: 24:36 · Proyecto 2: Ruleta de casino

Implementamos un modo trampa para demostraciones y pruebas: predecir números (simulado), cambiar el saldo, y banderas de depuración para controlar el comportamiento del juego.

Modo trampa: predecir números

Simulamos "predicción" del próximo número (en realidad es solo el generador normal):

MODO_TRAMPA=false

activar_modo_trampa() {
  MODO_TRAMPA=true
  echo "${MAGENTA}${BOLD}⚠️ MODO TRAMPA ACTIVADO${RESET}"
  echo "• Puedes predecir números (simulado)"
  echo "• Puedes cambiar el saldo"
  echo "• Las estadísticas no cuentan para logros"
  sleep 1
}

predecir_numero() {
  if [[ "$MODO_TRAMPA" == false ]]; then
    print_error "Modo trampa no activado"
    return 1
  fi

  # Simulación: mostraremos los próximos 3 números
  echo "${MAGENTA}Predicción de próxima jugada:${RESET}"
  for (( i = 1; i <= 3; i++ )); do
    local num color
    num=$(( RANDOM % 37 ))
    color=$(determinar_color "$num")
    printf "  %d. %s %d (%s)\n" "$i" "$([ "$color" = "rojo" ] && echo "$ROJO" || ([ "$color" = "verde" ] && echo "$VERDE") || echo "$BLANCO")●${RESET}" "$num" "$color"
  done
}

Cambiar saldo para pruebas

cambiar_saldo() {
  if [[ "$MODO_TRAMPA" == false ]]; then
    print_error "Modo trampa no activado"
    return
  fi

  echo "Saldo actual: ${VERDE}\$${SALDO}${RESET}"
  read -r -p "Nuevo saldo: " nuevo_saldo

  if ! [[ "$nuevo_saldo" =~ ^[0-9]+$ ]] || [[ "$nuevo_saldo" -le 0 ]]; then
    print_error "Saldo inválido"
    return
  fi

  local diferencia=$(( nuevo_saldo - SALDO ))
  SALDO=$nuevo_saldo

  if [[ "$diferencia" -gt 0 ]]; then
    echo "${VERDE}Saldo incrementado en +\$${diferencia}${RESET}"
  else
    echo "${ROJO}Saldo reducido en ${diferencia}${RESET}"
  fi

  [[ "$SALDO" -gt "$MAX_SALDO" ]] && MAX_SALDO=$SALDO
}

Forzar resultado específico

Para probar casos borde, forzamos un número concreto:

FORZAR_RESULTADO=-1

forzar_numero() {
  if [[ "$MODO_TRAMPA" == false ]]; then
    print_error "Modo trampa no activado"
    return
  fi

  read -r -p "Número a forzar (0-36, -1 para desactivar): " num

  if [[ "$num" -eq -1 ]]; then
    FORZAR_RESULTADO=-1
    print_info "Forzado desactivado"
    return
  fi

  if ! [[ "$num" =~ ^[0-9]+$ ]] || [[ "$num" -lt 0 ]] || [[ "$num" -gt 36 ]]; then
    print_error "Número inválido (0-36)"
    return
  fi

  FORZAR_RESULTADO=$num
  echo "${MAGENTA}${BOLD}⚠️ Próximo número forzado: $num${RESET}"
}

# Modificar girar_ruleta para usar forzado
girar_ruleta() {
  local numero

  if [[ "$FORZAR_RESULTADO" -ge 0 ]]; then
    numero=$FORZAR_RESULTADO
    FORZAR_RESULTADO=-1  # Solo una vez
  else
    numero=$(( RANDOM % 37 ))
  fi

  local color paridad
  color=$(determinar_color "$numero")
  paridad=$(determinar_paridad "$numero")

  ULTIMO_NUMERO=$numero
  ULTIMO_COLOR=$color
  ULTIMA_PARIDAD=$paridad

  echo "$numero|$color|$paridad"
}

Banderas de depuración

Añadimos opciones de línea de comandos para controlar el modo trampa:

# Opciones de línea de comandos
while getopts "s:t:d:h" opt; do
  case $opt in
    s) SALDO="$OPTARG" ;;
    t) MODO_TRAMPA=true ;;
    d) DEBUG_MODE=true ;;
    h)
      echo "Uso: $0 [opciones]"
      echo "  -s    Saldo inicial"
      echo "  -t           Activar modo trampa"
      echo "  -d           Activar modo debug (set -x)"
      echo "  -h           Mostrar ayuda"
      exit 0
      ;;
    *) echo "Opción inválida"; exit 1 ;;
  esac
done

if [[ "$DEBUG_MODE" == true ]]; then
  set -x  # Mostrar cada comando antes de ejecutarlo
fi

Menú de depuración

menu_debug() {
  while true; do
    echo "${BOLD}${MAGENTA}══════ MENÚ DEBUG ══════${RESET}"
    select opt in "Activar/Desactivar trampa" "Predecir números" "Cambiar saldo" "Forzar número" "Ver variables" "Volver"; do
      case $opt in
        "Activar/Desactivar trampa")
          if [[ "$MODO_TRAMPA" == true ]]; then
            MODO_TRAMPA=false
            echo "Modo trampa desactivado"
          else
            activar_modo_trampa
          fi
          break
          ;;
        "Predecir números") predecir_numero; break ;;
        "Cambiar saldo") cambiar_saldo; break ;;
        "Forzar número") forzar_numero; break ;;
        "Ver variables")
          echo "SALDO=$SALDO APUESTA=$APUESTA TRAMPA=$MODO_TRAMPA FORZADO=$FORZAR_RESULTADO"
          echo "JUGADAS=$TOTAL_JUGADAS GANADAS=$TOTAL_GANADAS PERDIDAS=$TOTAL_PERDIDAS"
          break
          ;;
        "Volver") return ;;
      esac
    done
  done
}
