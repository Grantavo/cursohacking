Scripting en Bash [10-15]

Duración: 22:43 · Proyecto 2: Ruleta de casino

Añadimos sonidos con el altavoz del sistema (beep), notificaciones visuales, un modo silencioso (-s) y la capacidad de personalizar colores.

Sonidos con beep

Usamos el bell de la terminal y speaker-test para efectos de sonido básicos:

SONIDO_ACTIVADO=true

beep() {
  if [[ "$SONIDO_ACTIVADO" == false ]]; then
    return
  fi
  echo -ne "\a"  # Bell character
}

sonido_ganar() {
  if [[ "$SONIDO_ACTIVADO" == false ]]; then
    return
  fi
  # Tonos ascendentes
  for freq in 500 600 700 800 900 1000; do
    ( speaker-test -t sine -f "$freq" -l 1 2>/dev/null & )
    sleep 0.05
  done
}

sonido_perder() {
  if [[ "$SONIDO_ACTIVADO" == false ]]; then
    return
  fi
  # Tonos descendentes
  for freq in 400 300 200 150; do
    ( speaker-test -t sine -f "$freq" -l 1 2>/dev/null & )
    sleep 0.1
  done
}

sonido_giro() {
  if [[ "$SONIDO_ACTIVADO" == false ]]; then
    return
  fi
  # Sonido de clic rápido simulado
  for (( i = 0; i < 10; i++ )); do
    beep
    sleep 0.05
  done
}
⚠️ speaker-test: Requiere el módulo de sonido del sistema. En entornos sin altavoz o en SSH, no producirá sonido. Úsalo solo en tu máquina local con escritorio.

Notificaciones visuales

notificar() {
  local mensaje=$1
  local tipo=$2  # "info", "exito", "error"

  case $tipo in
    exito)
      echo "${VERDE}${BOLD}[✓]${RESET} ${VERDE}${mensaje}${RESET}"
      sonido_ganar
      ;;
    error)
      echo "${ROJO}${BOLD}[✗]${RESET} ${ROJO}${mensaje}${RESET}"
      sonido_perder
      ;;
    info)
      echo "${AZUL}${BOLD}[i]${RESET} ${AZUL}${mensaje}${RESET}"
      beep
      ;;
    warning)
      echo "${AMARILLO}${BOLD}[⚠]${RESET} ${AMARILLO}${mensaje}${RESET}"
      beep
      ;;
  esac
}

notificacion_ganancia() {
  local cantidad=$1
  local tipo=$2

  if [[ "$cantidad" -gt 1000 ]]; then
    notificar "🔥 ¡GANANCIA ÉPICA! +\$${cantidad}" "exito"
  elif [[ "$cantidad" -gt 500 ]]; then
    notificar "⭐ ¡Gran ganada! +\$${cantidad}" "exito"
  elif [[ "$cantidad" -gt 0 ]]; then
    notificar "✓ Ganaste +\$${cantidad}" "exito"
  fi
}

Modo silencioso (-s)

Opción para desactivar todos los sonidos:

# En el procesamiento de argumentos
SILENT_MODE=false

while getopts "s:t:d:h" opt; do
  case $opt in
    s) SILENT_MODE=true ;;
    # ... otros casos
  esac
done

# Al inicio del juego
if [[ "$SILENT_MODE" == true ]]; then
  SONIDO_ACTIVADO=false
  notificar "Modo silencioso activado" "info"
fi

Colores personalizables

Permitimos al usuario definir sus propios colores mediante variables de entorno o archivo de configuración:

# Configuración de colores personalizables
COLOR_PRINCIPAL="${COLOR_PRINCIPAL:-$AMARILLO}"
COLOR_SALDO="${COLOR_SALDO:-$VERDE}"
COLOR_ALERTA="${COLOR_ALERTA:-$ROJO}"
COLOR_INFO="${COLOR_INFO:-$AZUL}"
COLOR_EXITO="${COLOR_EXITO:-$VERDE}"

cargar_configuracion_colores() {
  local config_file="${HOME}/.roulette/colors.conf"

  if [[ -f "$config_file" ]]; then
    source "$config_file"
    notificar "Colores personalizados cargados" "info"
  fi
}

Uso de tput para temas

También podemos ofrecer temas predefinidos:

aplicar_tema() {
  local tema=$1

  case $tema in
    oscuro)
      COLOR_PRINCIPAL=$CIAN
      COLOR_SALDO=$VERDE
      COLOR_ALERTA=$ROJO
      ;;
    claro)
      COLOR_PRINCIPAL=$MAGENTA
      COLOR_SALDO=$AZUL
      COLOR_ALERTA=$AMARILLO
      ;;
    hacker)
      COLOR_PRINCIPAL=$VERDE
      COLOR_SALDO=$VERDE
      COLOR_ALERTA=$ROJO
      ;;
    *)
      print_error "Tema desconocido: $tema"
      return 1
      ;;
  esac

  print_success "Tema '${tema}' aplicado"
}
💡 tput vs códigos ANSI: tput es más portable que escribir códigos ANSI directamente (\e[31m). Funciona en cualquier terminal que soporte terminfo.
