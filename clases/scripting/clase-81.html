Scripting en Bash [11-15]

Duración: 17:41 · Proyecto 2: Ruleta de casino

Implementamos modo multijugador local con turnos alternados, rankings, archivo de puntuaciones y funciones de ordenación.

Estructura multijugador

declare -A JUGADORES  # Array asociativo: nombre -> saldo
declare -A TURNOS
JUGADOR_ACTUAL=""
RONDAS_MULTI=5

inicializar_multijugador() {
  read -r -p "Número de jugadores (2-4): " num_jugadores
  num_jugadores="${num_jugadores:-2}"

  if ! [[ "$num_jugadores" =~ ^[2-4]$ ]]; then
    print_error "Deben ser 2-4 jugadores"
    return 1
  fi

  for (( i = 1; i <= num_jugadores; i++ )); do
    read -r -p "Nombre del jugador $i: " nombre
    nombre="${nombre:-Jugador$i}"
    JUGADORES[$nombre]=1000
  done

  echo
  echo "${BOLD}Jugadores:${RESET}"
  for nombre in "${!JUGADORES[@]}"; do
    echo "  • ${nombre}: ${VERDE}\$${JUGADORES[$nombre]}${RESET}"
  done

  JUGADOR_ACTUAL="${!JUGADORES[@]}"  # Primer jugador
  sleep 1
}

Turnos alternados

turno_multijugador() {
  local nombres=()
  for nombre in "${!JUGADORES[@]}"; do
    nombres+=("$nombre")
  done

  local turno_actual=0
  local ronda=0

  while [[ "$ronda" -lt "$RONDAS_MULTI" ]]; do
    local jugador="${nombres[$turno_actual]}"
    local saldo=${JUGADORES[$jugador]}

    clear
    echo "${BOLD}${AZUL}═══ Ronda $((ronda+1))/${RONDAS_MULTI} — Turno de: ${jugador} ═══${RESET}"
    echo "Saldo: ${VERDE}\$${saldo}${RESET}"
    echo

    if [[ "$saldo" -le 0 ]]; then
      echo "${ROJO}${jugador} está en bancarrota y se salta el turno${RESET}"
      sleep 1
    else
      # Copiar saldo al jugador global
      SALDO=$saldo
      NOMBRE=$jugador

      # Ejecutar un turno normal
      realizar_apuesta

      # Guardar saldo de vuelta
      JUGADORES[$jugador]=$SALDO
    fi

    # Siguiente jugador
    turno_actual=$(( (turno_actual + 1) % ${#nombres[@]} ))
    if [[ "$turno_actual" -eq 0 ]]; then
      (( ronda++ ))
    fi
  done

  mostrar_ranking
}

Archivo de puntuaciones

Guardamos las puntuaciones máximas históricas:

SCORE_FILE="${GUARDAR_DIR}/scores.txt"

guardar_puntuacion() {
  local nombre=$1
  local saldo=$2
  local fecha
  fecha=$(date '+%Y-%m-%d %H:%M')

  mkdir -p "$GUARDAR_DIR"
  echo "${fecha}|${nombre}|${saldo}" >> "$SCORE_FILE"

  notificar "Puntuación guardada: ${nombre} - \$${saldo}" "info"
}

cargar_puntuaciones() {
  if [[ ! -f "$SCORE_FILE" ]]; then
    return
  fi

  # Usar sort para ordenar por saldo descendente
  sort -t'|' -k3 -rn "$SCORE_FILE" | head -10
}

Funciones de ordenación

Ordenamos jugadores por saldo para mostrar el ranking:

ordenar_jugadores() {
  # Usar sort externo para ordenar arrays asociativos
  for nombre in "${!JUGADORES[@]}"; do
    echo "${JUGADORES[$nombre]}|${nombre}"
  done | sort -t'|' -k1 -rn
}

mostrar_ranking() {
  clear
  echo "${BOLD}${AMARILLO}═══════════════════════════════════${RESET}"
  echo "${BOLD}          🏆 RANKING FINAL${RESET}"
  echo "${BOLD}${AMARILLO}═══════════════════════════════════${RESET}"
  echo

  local pos=1
  ordenar_jugadores | while IFS='|' read -r saldo nombre; do
    local medalla
    case $pos in
      1) medalla="${AMARILLO}🥇${RESET}" ;;
      2) medalla="${BLANCO}🥈${RESET}" ;;
      3) medalla="${ROJO}🥉${RESET}" ;;
      *) medalla="   " ;;
    esac

    printf "%s %2d. %-20s ${VERDE}\$%d${RESET}\n" "$medalla" "$pos" "$nombre" "$saldo"
    (( pos++ ))
  done

  echo
  echo "${BOLD}Mejores puntuaciones históricas:${RESET}"
  cargar_puntuaciones | while IFS='|' read -r fecha nombre saldo; do
    printf "  [%s] %-20s \$${saldo}\n" "$fecha" "$nombre"
  done
}

Game over multijugador

finalizar_multijugador() {
  mostrar_ranking

  # Guardar puntuaciones
  for nombre in "${!JUGADORES[@]}"; do
    guardar_puntuacion "$nombre" "${JUGADORES[$nombre]}"
  done

  echo
  read -n 1 -s -r -p "Presiona cualquier tecla para salir..."
  exit 0
}
💡 sort con delimitadores: sort -t'|' -k3 -rn ordena usando | como separador de campos (-t), por el tercer campo (-k3), en orden numérico inverso (-rn). Muy útil para ordenar datos en Bash.
