Scripting en Bash [12-15]

Duración: 18:36 · Proyecto 2: Ruleta de casino

Implementamos un sistema de logros y desbloqueables: el jugador gana logros al completar hitos, sube de nivel, recibe bonus por rachas y puede personalizar su perfil.

Sistema de logros

Cada logro tiene un nombre, descripción y condición de desbloqueo:

declare -a LOGROS_DESBLOQUEADOS=()

declare -A LOGRO_NOMBRE
LOGRO_NOMBRE[primera_apuesta]="Primera Apuesta"
LOGRO_NOMBRE[ganar_primera]="Primera Victoria"
LOGRO_NOMBRE[racha_3]="Racha de 3"
LOGRO_NOMBRE[racha_5]="Racha de 5"
LOGRO_NOMBRE[bankroll_5000]="Bankroll 5000"
LOGRO_NOMBRE[bankroll_10000]="Bankroll 10000"
LOGRO_NOMBRE[pleno_acertado]="Pleno Acertado"
LOGRO_NOMBRE[bancarrota]="Bancarrota"
LOGRO_NOMBRE[100_jugadas]="Veterano"

declare -A LOGRO_DESC
LOGRO_DESC[primera_apuesta]="Haz tu primera apuesta"
LOGRO_DESC[ganar_primera]="Gana tu primera partida"
LOGRO_DESC[racha_3]="Gana 3 veces seguidas"
LOGRO_DESC[racha_5]="Gana 5 veces seguidas"
LOGRO_DESC[bankroll_5000]="Alcanza $5000 de saldo"
LOGRO_DESC[bankroll_10000]="Alcanza $10000 de saldo"
LOGRO_DESC[pleno_acertado]="Acierta un número pleno"
LOGRO_DESC[bancarrota]="Pierde todo tu saldo"
LOGRO_DESC[100_jugadas]="Completa 100 jugadas"

verificar_logros() {
  # Primera apuesta
  if [[ "$TOTAL_JUGADAS" -ge 1 ]] && ! [[ " ${LOGROS_DESBLOQUEADOS[*]} " =~ "primera_apuesta" ]]; then
    desbloquear_logro "primera_apuesta"
  fi

  # Primera victoria
  if [[ "$TOTAL_GANADAS" -ge 1 ]] && ! [[ " ${LOGROS_DESBLOQUEADOS[*]} " =~ "ganar_primera" ]]; then
    desbloquear_logro "ganar_primera"
  fi

  # Bankroll
  if [[ "$SALDO" -ge 5000 ]] && ! [[ " ${LOGROS_DESBLOQUEADOS[*]} " =~ "bankroll_5000" ]]; then
    desbloquear_logro "bankroll_5000"
  fi
  if [[ "$SALDO" -ge 10000 ]] && ! [[ " ${LOGROS_DESBLOQUEADOS[*]} " =~ "bankroll_10000" ]]; then
    desbloquear_logro "bankroll_10000"
  fi

  # 100 jugadas
  if [[ "$TOTAL_JUGADAS" -ge 100 ]] && ! [[ " ${LOGROS_DESBLOQUEADOS[*]} " =~ "100_jugadas" ]]; then
    desbloquear_logro "100_jugadas"
  fi
}

desbloquear_logro() {
  local logro_id=$1
  LOGROS_DESBLOQUEADOS+=("$logro_id")

  echo
  echo "${AMARILLO}${BOLD}═══════════════════════════════${RESET}"
  echo "${BOLD}🏆 ¡LOGO DESBLOQUEADO!${RESET}"
  echo "${AMARILLO}${BOLD}═══════════════════════════════${RESET}"
  echo "${BOLD}${LOGRO_NOMBRE[$logro_id]}${RESET}"
  echo "${LOGRO_DESC[$logro_id]}"
  sonido_ganar
  sleep 1
}

Sistema de niveles

El nivel del jugador sube con la experiencia acumulada:

NIVEL=1
EXPERIENCIA=0
SIGUIENTE_NIVEL=100

calcular_experiencia() {
  local ganancia=$1
  local exp=$(( ganancia / 10 ))
  [[ "$exp" -lt 1 ]] && exp=1
  echo "$exp"
}

sumar_experiencia() {
  local exp=$1
  EXPERIENCIA=$(( EXPERIENCIA + exp ))

  while [[ "$EXPERIENCIA" -ge "$SIGUIENTE_NIVEL" ]]; do
    (( NIVEL++ ))
    EXPERIENCIA=$(( EXPERIENCIA - SIGUIENTE_NIVEL ))
    SIGUIENTE_NIVEL=$(( SIGUIENTE_NIVEL + 50 ))
    echo "${AMARILLO}${BOLD}⬆ ¡Has subido al nivel ${NIVEL}!${RESET}"
    sonido_ganar

    # Bonus por subir de nivel
    local bonus=$(( NIVEL * 50 ))
    SALDO=$(( SALDO + bonus ))
    echo "${VERDE}Bonus de nivel: +\$${bonus}${RESET}"
    sleep 1
  done
}

barra_experiencia() {
  local porcentaje=$(( EXPERIENCIA * 100 / SIGUIENTE_NIVEL ))
  local ancho=20
  local lleno=$(( porcentaje * ancho / 100 ))

  echo -n "Nv.${NIVEL} ["
  for (( i = 0; i < ancho; i++ )); do
    if [[ "$i" -lt "$lleno" ]]; then
      echo -n "${CIAN}■${RESET}"
    else
      echo -n "·"
    fi
  done
  echo "] ${porcentaje}%"
}

Bonus por racha

CALLE_RAFA=0
MAX_RACHA_VICTORIAS=0
RACHA_VICTORIAS_ACTUAL=0

verificar_racha() {
  local resultado=$1

  if [[ "$resultado" == "ganada" ]]; then
    (( RACHA_VICTORIAS_ACTUAL++ ))
    [[ "$RACHA_VICTORIAS_ACTUAL" -gt "$MAX_RACHA_VICTORIAS" ]] && MAX_RACHA_VICTORIAS=$RACHA_VICTORIAS_ACTUAL

    # Bonus por racha
    if [[ "$RACHA_VICTORIAS_ACTUAL" -eq 3 ]] && ! [[ " ${LOGROS_DESBLOQUEADOS[*]} " =~ "racha_3" ]]; then
      desbloquear_logro "racha_3"
      SALDO=$(( SALDO + 100 ))
      echo "${VERDE}Bonus racha 3: +\$100${RESET}"
    fi
    if [[ "$RACHA_VICTORIAS_ACTUAL" -eq 5 ]] && ! [[ " ${LOGROS_DESBLOQUEADOS[*]} " =~ "racha_5" ]]; then
      desbloquear_logro "racha_5"
      SALDO=$(( SALDO + 300 ))
      echo "${VERDE}Bonus racha 5: +\$300${RESET}"
    fi

    # Bonificación progresiva por racha
    if [[ "$RACHA_VICTORIAS_ACTUAL" -gt 5 ]]; then
      local extra=$(( RACHA_VICTORIAS_ACTUAL * 20 ))
      SALDO=$(( SALDO + extra ))
      echo "${CIAN}🔥 Racha de ${RACHA_VICTORIAS_ACTUAL}: +\$${extra} bonus${RESET}"
    fi
  else
    RACHA_VICTORIAS_ACTUAL=0
  fi
}

Personalización del perfil

mostrar_perfil() {
  clear
  echo "${BOLD}${AZUL}═══════════════════════════════════${RESET}"
  echo "${BOLD}         👤 PERFIL DE JUGADOR${RESET}"
  echo "${BOLD}${AZUL}═══════════════════════════════════${RESET}"
  echo
  echo "Nombre:     ${BOLD}${NOMBRE}${RESET}"
  echo "Nivel:      ${BOLD}${NIVEL}${RESET}"
  barra_experiencia
  echo "Saldo:      ${VERDE}\$${SALDO}${RESET}"
  echo "Jugadas:    $TOTAL_JUGADAS"
  echo "Ganadas:    ${VERDE}$TOTAL_GANADAS${RESET}"
  echo "Perdidas:   ${ROJO}$TOTAL_PERDIDAS${RESET}"
  echo "Max racha:  $MAX_RACHA_VICTORIAS"
  echo
  echo "${BOLD}🏆 Logros (${#LOGROS_DESBLOQUEADOS[@]}):${RESET}"
  if [[ "${#LOGROS_DESBLOQUEADOS[@]}" -eq 0 ]]; then
    echo "  Aún no tienes logros. ¡Sigue jugando!"
  else
    for logro_id in "${LOGROS_DESBLOQUEADOS[@]}"; do
      echo "  ${AMARILLO}★${RESET} ${LOGRO_NOMBRE[$logro_id]}"
    done
  fi
  echo
  read -n 1 -s -r -p "Presiona cualquier tecla para continuar..."
}
