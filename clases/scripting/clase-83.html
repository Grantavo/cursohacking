Ejemplo de función bien documentada:

# === roulette_core.sh ===

# generar_numero()
# Genera un número aleatorio entre 0 y 36.
# Usa $RANDOM por defecto, o /dev/urandom si está disponible.
# Argumentos: ninguno
# Retorna: número entre 0 y 36 (stdout)
generar_numero() {
  if [[ -r /dev/urandom ]]; then
    local byte
    byte=$(od -An -N1 -tu1 /dev/urandom | tr -d ' ')
    echo $(( byte % 37 ))
  else
    echo $(( RANDOM % 37 ))
  fi
}

# determinar_color(numero)
# Determina el color de un número de la ruleta europea.
# Argumentos:
#   $1 - Número (0-36)
# Retorna: "rojo", "negro" o "verde" (stdout)
determinar_color() {
  local -r num=$1
  local -ra ROJOS=(1 3 5 7 9 12 14 16 18 19 21 23 25 27 30 32 34 36)

  [[ "$num" -eq 0 ]] && { echo "verde"; return 0; }

  for n in "${ROJOS[@]}"; do
    [[ "$n" -eq "$num" ]] && { echo "rojo"; return 0; }
  done

  echo "negro"
}

# Usar declare y readonly para claridad y seguridad
procesar_apuesta_refactor() {
  local -r tipo=$1
  local -r eleccion=$2
  local -r cantidad=$3
  local -r resultado=$4
  local ganancia=0

  case $tipo in
    pleno)
      if [[ "$eleccion" -eq "$resultado" ]]; then
        ganancia=$(( cantidad * 35 ))
      fi
      ;;
    docena)
      local -r min=$(( (eleccion - 1) * 12 + 1 ))
      local -r max=$(( eleccion * 12 ))
      if (( resultado >= min && resultado <= max )); then
        ganancia=$(( cantidad * 2 ))
      fi
      ;;
    color|mitad|par_impar)
      # Delegar a funciones específicas
      ganancia=$(procesar_apuesta_simple "$tipo" "$eleccion" "$cantidad" "$resultado")
      ;;
  esac

  echo "$ganancia"
}

Principios aplicados

# 1. DRY (Don't Repeat Yourself)
# Antes: código duplicado en cada tipo de apuesta
# Después: función genérica con parámetros

# 2. SRP (Single Responsibility Principle)
# Cada archivo tiene UNA responsabilidad clara
# ui.sh -> solo interfaz
# bets.sh -> solo lógica de apuestas

# 3. KISS (Keep It Simple, Stupid)
# Funciones pequeñas (< 20 líneas)
# Nombres descriptivos en inglés o español consistente

# 4. Separación de configuración y lógica
# config.sh contiene solo variables
# El resto del código solo las referencia
📐 Refactorización: Dividir el código en módulos no solo lo hace más mantenible, sino que permite trabajar en equipo (cada persona en un archivo) y reutilizar funciones en otros proyectos.
