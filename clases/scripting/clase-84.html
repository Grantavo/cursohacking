Scripting en Bash [14-15]

Duración: 33:20 · Proyecto 2: Ruleta de casino

Cubrimos pruebas y depuración: set -x y set -e para tracing, trap para limpieza, pruebas unitarias básicas con assert casero y manejo de casos borde.

Modo debug con set -x y set -e

# Activar desde CLI: ./roulette.sh -d

debug_mode() {
  echo "${AMARILLO}Modo debug activado${RESET}"
  set -x  # Muestra cada comando antes de ejecutarlo
  set -e  # Sale si algún comando falla (exit code != 0)
  set -u  # Error si se usa variable no definida
  set -o pipefail  # Error si algún comando en pipe falla
}

# Fragmento de depuración selectiva
depurar_funcion() {
  set -x  # Activar trace solo para esta función
  # ... código a depurar ...
  set +x  # Desactivar trace
}

# Ejemplo de depuración de getopts
depurar_args() {
  echo "ARGV: $@"
  echo "ARGC: $#"
  for (( i = 0; i < $#; i++ )); do
    echo "  ARG[$i] = ${!i}"
  done
}

Trap para limpieza

Garantizamos que los recursos temporales se limpien incluso si el script falla:

cleanup() {
  local exit_code=$?
  echo
  echo "${CIAN}Limpiando recursos...${RESET}"

  # Eliminar archivos temporales
  [[ -f /tmp/roulette_historial.tmp ]] && rm -f /tmp/roulette_historial.tmp
  [[ -f /tmp/roulette_debug.log ]] && rm -f /tmp/roulette_debug.log

  # Restaurar configuración de terminal
  stty echo 2>/dev/null || true
  tput cnorm 2>/dev/null || true  # Mostrar cursor

  if [[ "$exit_code" -ne 0 ]]; then
    echo "${ROJO}El script terminó con error (código: $exit_code)${RESET}"
  else
    echo "${VERDE}Script finalizado correctamente${RESET}"
  fi

  exit "$exit_code"
}

# Registrar trap para señales comunes
trap cleanup EXIT INT TERM HUP

# Handler para Ctrl+C específico
ctrl_c() {
  echo
  echo "${AMARILLO}${BOLD}⏹ Juego interrumpido por el usuario${RESET}"
  guardar_partida  # Guardar antes de salir
  exit 130
}
trap ctrl_c SIGINT

Pruebas unitarias básicas

Creamos un framework de pruebas minimalista:

# === tests/test_roulette.sh ===

PASADAS=0
FALLADAS=0

assert() {
  local descripcion=$1
  local esperado=$2
  local resultado=$3

  if [[ "$resultado" == "$esperado" ]]; then
    echo "${VERDE}[✓]${RESET} $descripcion"
    (( PASADAS++ ))
  else
    echo "${ROJO}[✗]${RESET} $descripcion"
    echo "  Esperado: '$esperado'"
    echo "  Obtenido: '$resultado'"
    (( FALLADAS++ ))
  fi
}

test_determinar_color() {
  echo "${BOLD}Test: determinar_color${RESET}"

  assert "0 es verde" "verde" "$(determinar_color 0)"
  assert "1 es rojo" "rojo" "$(determinar_color 1)"
  assert "2 es negro" "negro" "$(determinar_color 2)"
  assert "15 es negro" "negro" "$(determinar_color 15)"
  assert "19 es rojo" "rojo" "$(determinar_color 19)"
}

test_determinar_paridad() {
  echo "${BOLD}Test: determinar_paridad${RESET}"

  assert "0 es ninguno" "ninguno" "$(determinar_paridad 0)"
  assert "2 es par" "par" "$(determinar_paridad 2)"
  assert "3 es impar" "impar" "$(determinar_paridad 3)"
}

test_apuesta_pleno() {
  echo "${BOLD}Test: apuesta pleno${RESET}"

  local ganancia
  ganancia=$(procesar_apuesta "pleno" "17" "10" "17")
  assert "Pleno acertado paga 35:1" 350 "$ganancia"

  ganancia=$(procesar_apuesta "pleno" "17" "10" "5")
  assert "Pleno fallado paga 0" 0 "$ganancia"
}

test_apuesta_docena() {
  echo "${BOLD}Test: apuesta docena${RESET}"

  local ganancia
  ganancia=$(procesar_apuesta "docena" "1" "10" "5")
  assert "Docena 1 acertada paga 2:1" 20 "$ganancia"

  ganancia=$(procesar_apuesta "docena" "1" "10" "20")
  assert "Docena 1 fallada paga 0" 0 "$ganancia"
}

# Ejecutar todas las pruebas
run_tests() {
  echo "${BOLD}${AZUL}══════ PRUEBAS UNITARIAS ══════${RESET}"
  echo

  test_determinar_color
  echo
  test_determinar_paridad
  echo
  test_apuesta_pleno
  echo
  test_apuesta_docena
  echo
  test_apuesta_color

  echo
  echo "${BOLD}Resumen:${RESET} ${VERDE}$PASADAS pasadas${RESET}, ${ROJO}$FALLADAS falladas${RESET}"
  [[ "$FALLADAS" -eq 0 ]] && echo "${VERDE}${BOLD}¡Todas las pruebas pasaron!${RESET}"
}
⚠️ set -e con cuidado: set -e puede ser demasiado agresivo. Comandos que devuelven código de salida distinto de 0 en situaciones normales (como grep sin coincidencias) pueden detener el script inesperadamente.

Casos borde

test_casos_borde() {
  echo "${BOLD}Test: casos borde${RESET}"

  # Saldo cero
  SALDO=0
  assert "Apuesta con saldo cero no debería ocurrir" 0 \
    "$(validar_cantidad 10 2>/dev/null; echo $?)"

  # Número fuera de rango
  assert "Número 37 es inválido" 1 \
    "$(validar_numero_ruleta 37 2>/dev/null; echo $?)"

  # Número negativo
  assert "Número -1 es inválido" 1 \
    "$(validar_numero_ruleta -1 2>/dev/null; echo $?)"

  # String vacío
  assert "String vacío es inválido" 1 \
    "$(validar_cantidad '' 2>/dev/null; echo $?)"

  # Caracteres no numéricos
  assert "Texto no es válido" 1 \
    "$(validar_cantidad 'abc' 2>/dev/null; echo $?)"

  # Apuesta mayor que saldo
  SALDO=100
  assert "Apuesta > saldo es inválida" 1 \
    "$(validar_cantidad 200 2>/dev/null; echo $?)"

  # Resultado del 0 en apuesta de color
  local ganancia
  ganancia=$(procesar_apuesta "color" "rojo" "10" "0")
  assert "Color con 0 paga 0" 0 "$ganancia"
}
💡 Pruebas en Bash: No necesitas un framework complejo. Con funciones assert simples y test_* puedes cubrir la mayoría de casos. Para proyectos más grandes, considera bats (Bash Automated Testing System).
